05Modul 01 – Trainingsdaten sammeln

Dieses Skript nutzt die Pi Camera 3, um Fotos von Flaschen aus der Maschinenperspektive aufzunehmen. Ziel: mindestens 300–500 Bilder in verschiedenen Lichtverhältnissen und mit verschiedenen Flaschen.

📸 Empfohlene Vielfalt
  • Verschiedene Flaschentypen
  • Verschiedene Füllstände
  • 3+ Lichtverhältnisse
  • Leicht versetzt stellen
🏷️ Labeling-Tool
  • Roboflow (empfohlen)
  • Klasse: bottle_opening
  • Format: YOLOv8 (.txt)
  • Export: YOLO-Ordner
💡 Maker-Hack
  • Gantry in Nullposition fahren
  • Kamera-Feed live nutzen
  • Delay 0,5 s genügt
  • ~10 min für 300 Bilder
ai/01_collect_data.py 

#!/usr/bin/env python3
"""
Smart Beverage Station — Schritt 1: Trainingsdaten sammeln
Nimmt Fotos via Pi Camera 3 auf und speichert sie für das spätere Labeling.
Läuft direkt auf dem Raspberry Pi 5.
"""
import os
import time
import argparse
from datetime import datetime
from picamera2 import Picamera2


def collect_images(output_dir: str, n_images: int, delay: float = 0.5) -> None:
    """Nimmt `n_images` Bilder mit der Pi Camera auf und speichert sie in `output_dir`."""
    os.makedirs(output_dir, exist_ok=True)

    cam = Picamera2()
    config = cam.create_still_configuration(
        main={"size": (1280, 720), "format": "RGB888"},
        controls={"AwbEnable": True, "AeEnable": True},   # Auto-WB + Auto-Expo
    )
    cam.configure(config)
    cam.start()
    time.sleep(2.0)  # Kamera-Warmup (wichtig für korrekte Belichtung!)

    print(f"[INFO] Starte Aufnahme von {n_images} Bildern → '{output_dir}'")
    print("       Stelle die Flasche jetzt unter den Füllkopf.\n")

    saved = 0
    for i in range(n_images):
        ts       = datetime.now().strftime("%Y%m%d_%H%M%S_%f")
        filename = os.path.join(output_dir, f"bottle_{ts}.jpg")
        cam.capture_file(filename)
        saved   += 1
        print(f"  [{saved:04d}/{n_images}]  → {filename}")
        time.sleep(delay)

    cam.stop()
    print(f"\n✓  {saved} Bilder gespeichert in: {output_dir}")
    print(f"   Nächster Schritt: Bilder mit Roboflow labeln!")
    print(f"   → https://roboflow.com")


if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Trainingsdaten für YOLOv8 sammeln")
    parser.add_argument("--out",   type=str,   default="data/raw",
                        help="Ausgabeordner für Bilder (default: data/raw)")
    parser.add_argument("--n",     type=int,   default=300,
                        help="Anzahl aufzunehmender Bilder (default: 300)")
    parser.add_argument("--delay", type=float, default=0.5,
                        help="Pause zwischen Aufnahmen in Sekunden (default: 0.5)")
    args = parser.parse_args()

    collect_images(args.out, args.n, args.delay)
✅
Nach dem Sammeln: Lade die Bilder bei Roboflow.com hoch. Neue Klasse bottle_opening anlegen, Bounding Boxes um die Flaschenöffnungen zeichnen und als YOLOv8 Format exportieren.

06Modul 02 – Dataset vorbereiten

Dieses Skript teilt die gelabelten Rohdaten automatisch in Train / Val / Test auf (70 / 20 / 10 %) und erstellt die für YOLOv8 nötige data.yaml.

ai/02_prepare_dataset.py 

#!/usr/bin/env python3
"""
Smart Beverage Station — Schritt 2: Dataset für YOLOv8 vorbereiten
Erwartet: data/raw/*.jpg  +  data/raw/*.txt  (YOLO-Labels aus Roboflow/LabelImg)

Erstellt:
  datasets/bottles/
    ├── data.yaml
    ├── train/{images,labels}/
    ├── val/{images,labels}/
    └── test/{images,labels}/
"""
import os
import shutil
import random
from pathlib import Path


# ─── Konfiguration ────────────────────────────────────────────────────────────
RAW_DIR  = Path("data/raw")          # Bilder + Labels hier
DATASET  = Path("datasets/bottles")  # Ausgabe
SPLIT    = (0.70, 0.20, 0.10)        # train / val / test
SEED     = 42
CLASSES  = ["bottle_opening"]        # Klassen (1 reicht fürs Projekt)
# ──────────────────────────────────────────────────────────────────────────────


def write_data_yaml(dataset_path: Path, classes: list[str]) -> None:
    """Schreibt die data.yaml, die YOLOv8 zum Training braucht."""
    content = f"""# Smart Beverage Station – YOLOv8 Dataset
path: {dataset_path.resolve()}
train: train/images
val:   val/images
test:  test/images

nc: {len(classes)}
names: {classes}
"""
    (dataset_path / "data.yaml").write_text(content)
    print(f"  ✓ data.yaml geschrieben → {dataset_path / 'data.yaml'}")


def split_and_copy(raw_dir: Path, dataset: Path, split: tuple, seed: int) -> None:
    """Liest gelabelte Bilder ein, mischelt, splittet und kopiert sie."""
    all_images = sorted(raw_dir.glob("*.jpg"))

    # Nur Bilder mit passender Label-Datei verwenden
    labeled = [
        img for img in all_images
        if (raw_dir / f"{img.stem}.txt").exists()
    ]

    if not labeled:
        raise FileNotFoundError(
            f"\n✗  Keine gelabelten Bilder in '{raw_dir}' gefunden.\n"
            "   Jedes Bild braucht eine gleichnamige .txt-Datei (YOLO-Format).\n"
            "   → Bitte zuerst mit Roboflow.com labeln!"
        )

    random.seed(seed)
    random.shuffle(labeled)

    n       = len(labeled)
    n_train = int(n * split[0])
    n_val   = int(n * split[1])

    splits = {
        "train": labeled[:n_train],
        "val":   labeled[n_train : n_train + n_val],
        "test":  labeled[n_train + n_val :],
    }

    print(f"\n  Gesamt: {n} gelabelte Bilder")
    for split_name, files in splits.items():
        img_dir = dataset / split_name / "images"
        lbl_dir = dataset / split_name / "labels"
        img_dir.mkdir(parents=True, exist_ok=True)
        lbl_dir.mkdir(parents=True, exist_ok=True)

        for img_path in files:
            lbl_path = raw_dir / f"{img_path.stem}.txt"
            shutil.copy2(img_path, img_dir / img_path.name)
            shutil.copy2(lbl_path, lbl_dir / f"{img_path.stem}.txt")

        print(f"  {split_name:5s}: {len(files):3d} Bilder")


if __name__ == "__main__":
    print(f"[INFO] Verarbeite Dataset aus: {RAW_DIR}")
    split_and_copy(RAW_DIR, DATASET, SPLIT, SEED)
    write_data_yaml(DATASET, CLASSES)
    print(f"\n✓  Dataset fertig unter: {DATASET}")
    print(f"   Nächster Schritt: python 03_train.py")

07Modul 03 – YOLOv8n Training

Trainiert ein YOLOv8n-Modell (Nano = kleinste Variante, schnellste Inferenz auf dem Pi). Muss auf dem PC mit RTX 4060 ausgeführt werden – Training auf CPU dauert ca. 50× länger.

⏱️ Trainingsdauer
RTX 4060: ~15–30 Min für 100 Epochen bei 300 Bildern. CPU: mehrere Stunden.
🎯 Ziel mAP50
Anstreben: > 0.85 mAP50. Mit 300+ Bildern und variiertem Labeling gut erreichbar.
💡 Early Stopping
Patience=20 Epochen stoppt das Training automatisch, wenn keine Verbesserung mehr messbar ist.
ai/03_train.py 

#!/usr/bin/env python3
"""
Smart Beverage Station — Schritt 3: YOLOv8n Training
Trainiert das Flaschenerkenner-Modell. Auf PC mit GPU ausführen!

  RTX 4060:  ~15-30 Minuten  ← empfohlen
  CPU only:  mehrere Stunden  ← nicht empfohlen

Voraussetzungen:
  pip install ultralytics torch torchvision --index-url https://download.pytorch.org/whl/cu121
"""
from pathlib import Path
from ultralytics import YOLO
import torch


# ─── Konfiguration ────────────────────────────────────────────────────────────
MODEL       = "yolov8n.pt"                    # Nano = schnellste Pi-Inferenz
DATA_YAML   = "datasets/bottles/data.yaml"
PROJECT_DIR = "runs/train"
RUN_NAME    = "sbs_bottle_v1"
EPOCHS      = 100
IMG_SIZE    = 640                             # Standard YOLO Input-Größe
BATCH_SIZE  = 16                              # Für RTX 4060; bei 8 GB VRAM
# ──────────────────────────────────────────────────────────────────────────────


def main() -> str:
    device = "0" if torch.cuda.is_available() else "cpu"
    if device == "cpu":
        print("⚠️  WARNUNG: Keine CUDA-GPU erkannt – Training auf CPU sehr langsam!")
    else:
        gpu = torch.cuda.get_device_name(0)
        print(f"✓  GPU erkannt: {gpu}")

    print(f"   Modell:  {MODEL}")
    print(f"   Daten:   {DATA_YAML}")
    print(f"   Epochen: {EPOCHS}  |  ImgSize: {IMG_SIZE}  |  Batch: {BATCH_SIZE}\n")

    model   = YOLO(MODEL)
    results = model.train(
        data      = DATA_YAML,
        epochs    = EPOCHS,
        imgsz     = IMG_SIZE,
        batch     = BATCH_SIZE,
        project   = PROJECT_DIR,
        name      = RUN_NAME,
        device    = device,

        # ── Daten-Augmentierung (verbessert Robustheit deutlich) ──
        hsv_h     = 0.015,   # Farbton-Variation
        hsv_s     = 0.7,     # Sättigungs-Variation
        hsv_v     = 0.4,     # Helligkeits-Variation  ← wichtig für Licht-Variationen
        degrees   = 10.0,    # Kleine Rotation (Flasche steht nie 100% gerade)
        translate = 0.1,     # Leichte Verschiebung
        scale     = 0.5,     # Zoomstufen-Variation
        fliplr    = 0.5,     # Horizontales Spiegeln
        mosaic    = 1.0,     # Mosaic-Augmentierung (4 Bilder kombiniert)

        # ── Optimierung ──────────────────────────────────────────
        optimizer = "AdamW",
        lr0       = 0.001,
        patience  = 20,      # Early Stopping nach 20 Epochen ohne Verbesserung
        save      = True,
        plots     = True,    # Trainings-Kurven als PNG speichern
        verbose   = True,
    )

    # ── Ergebnisse ausgeben ───────────────────────────────────────────────────
    best   = Path(PROJECT_DIR) / RUN_NAME / "weights" / "best.pt"
    map50  = results.results_dict.get("metrics/mAP50(B)",    0.0)
    map95  = results.results_dict.get("metrics/mAP50-95(B)", 0.0)
    prec   = results.results_dict.get("metrics/precision(B)", 0.0)
    recall = results.results_dict.get("metrics/recall(B)",    0.0)

    print("\n" + "─" * 50)
    print("✓  Training abgeschlossen!")
    print(f"   mAP@50:    {map50:.3f}  (Ziel: > 0.850)")
    print(f"   mAP@50-95: {map95:.3f}")
    print(f"   Precision: {prec:.3f}")
    print(f"   Recall:    {recall:.3f}")
    print(f"\n   Bestes Modell: {best}")
    print(f"\n   Nächster Schritt: python 04_export.py --model \"{best}\"")
    print("─" * 50)

    return str(best)


if __name__ == "__main__":
    best_model_path = main()

08Modul 04 – Modell-Export für Raspberry Pi

Konvertiert das trainierte .pt-Modell in ein für den Pi optimiertes Format. ncnn ist auf dem Raspberry Pi 5 ARM Cortex-A76 deutlich schneller als ONNX.

⚡ Warum ncnn?
ncnn ist ein auf ARM-CPUs spezialisiertes Inferenz-Framework (entwickelt von Tencent). Auf dem Pi 5 erreicht YOLOv8n mit ncnn ca. 8–12 FPS – mit ONNX nur ~3–5 FPS. Die KI nutzt alle 4 Cortex-A76 Kerne optimal aus.
ai/04_export.py 

#!/usr/bin/env python3
"""
Smart Beverage Station — Schritt 4: Modell für Raspberry Pi 5 exportieren

Format-Vergleich (YOLOv8n, 640px, Pi 5):
  .pt   (PyTorch)  → nur für Training, ~3 FPS auf Pi
  .onnx (ONNX)     → ~4-5 FPS, universell
  .ncnn (ncnn)     → ~8-12 FPS, ARM-optimiert  ← EMPFOHLEN

Exportiert auf dem PC; das ncnn-Modell wird dann auf den Pi kopiert.
"""
import argparse
import shutil
from pathlib import Path
from ultralytics import YOLO


# ─── Konfiguration ────────────────────────────────────────────────────────────
DEFAULT_MODEL  = "runs/train/sbs_bottle_v1/weights/best.pt"
DEFAULT_FORMAT = "ncnn"
DEFAULT_IMGSIZE = 640
OUTPUT_DIR     = Path("models")
# ──────────────────────────────────────────────────────────────────────────────


def export_and_benchmark(model_path: str, fmt: str, imgsz: int) -> None:
    """Exportiert das Modell und gibt Deployment-Infos aus."""
    model_path = Path(model_path)
    if not model_path.exists():
        raise FileNotFoundError(f"Modell nicht gefunden: {model_path}")

    print(f"[INFO] Exportiere: {model_path}")
    print(f"[INFO] Format:     {fmt}")
    print(f"[INFO] Bildgröße:  {imgsz}px\n")

    model = YOLO(str(model_path))

    if fmt == "ncnn":
        # Optimal für Raspberry Pi 5 (ARM Cortex-A76, keine FP16 SIMD!)
        exported_path = model.export(
            format   = "ncnn",
            imgsz    = imgsz,
            half     = False,   # Pi 5 hat kein natives FP16 → False lassen!
            optimize = True,    # ncnn-spezifische Optimierungen aktivieren
        )
    elif fmt == "onnx":
        exported_path = model.export(
            format   = "onnx",
            imgsz    = imgsz,
            opset    = 17,
            simplify = True,    # Modellgraph vereinfachen (schneller)
        )
    else:
        raise ValueError(f"Unbekanntes Format '{fmt}'. Wähle: ncnn | onnx")

    # In models/-Ordner verschieben für bessere Übersicht
    OUTPUT_DIR.mkdir(exist_ok=True)
    dest = OUTPUT_DIR / Path(exported_path).name
    if Path(exported_path).is_dir():
        if dest.exists():
            shutil.rmtree(dest)
        shutil.copytree(exported_path, dest)
    else:
        shutil.copy2(exported_path, dest)

    # Validierung auf Trainings-Dataset
    val_results = model.val(data="datasets/bottles/data.yaml", verbose=False)
    map50 = val_results.results_dict.get("metrics/mAP50(B)", 0.0)

    print("\n" + "─" * 55)
    print("✓  Export abgeschlossen!")
    print(f"   Exportierter Pfad: {dest}")
    print(f"   mAP50 (Validierung): {map50:.3f}")
    print()
    print("   Auf Pi kopieren (SSH/SCP):")
    print(f'   scp -r "{dest}" pi@raspberrypi.local:~/sbs/ai/models/')
    print()
    print("   Dann auf dem Pi starten:")
    print(f'   python3 05_inference.py --model "models/{dest.name}"')
    print("─" * 55)


if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="YOLOv8-Modell für Pi exportieren")
    parser.add_argument("--model",  default=DEFAULT_MODEL,
                        help="Pfad zur best.pt Datei")
    parser.add_argument("--format", default=DEFAULT_FORMAT,
                        choices=["ncnn", "onnx"],
                        help="Export-Format (default: ncnn)")
    parser.add_argument("--imgsz",  type=int, default=DEFAULT_IMGSIZE,
                        help="Eingabe-Bildgröße in Pixeln (default: 640)")
    args = parser.parse_args()

    export_and_benchmark(args.model, args.format, args.imgsz)

09Modul 05 – Live-Inferenz auf dem Pi

Das Herzstück des Systems: Liest kontinuierlich die Pi Camera 3 aus, erkennt Flaschenöffnungen, rechnet Pixel in Millimeter um und sendet G-Code-Befehle an den ESP32.

⚠️
Kalibrierung erforderlich! Die Konstante PX_PER_MM muss eingemessen werden. Flasche an bekannte Position stellen, messen wie viele Pixel einem Millimeter entsprechen, und den Wert anpassen.
ai/05_inference.py 

#!/usr/bin/env python3
"""
Smart Beverage Station — Schritt 5: Live-Inferenz auf dem Raspberry Pi 5

Vollständiger Workflow:
  1. Kamera-Frame aufnehmen
  2. YOLOv8 erkennt Flaschenöffnung
  3. Pixel-Koordinate → Millimeter-Offset berechnen
  4. G-Code an ESP32 senden (USB-Serial)
  5. Relais öffnet Ventil → Getränk fließt

Läuft auf dem Raspberry Pi 5 mit picamera2 + ultralytics.
"""
import time
import argparse
import numpy as np
import serial
from picamera2 import Picamera2
from ultralytics import YOLO


# ─── Konfiguration (hier anpassen!) ───────────────────────────────────────────
MODEL_PATH   = "models/sbs_bottle_v1_ncnn_model"  # ncnn-Ordner vom Pi
SERIAL_PORT  = "/dev/ttyUSB0"   # ESP32 via USB (ggf. ttyACM0 testen)
BAUD_RATE    = 115200
CONF_THRESH  = 0.70             # Nur Erkennungen > 70% Konfidenz verwenden
IMG_W, IMG_H = 1280, 720        # Kamera-Auflösung

# !! KALIBRIERUNG: Diese Werte müssen eingemessen werden !!
# Messe: Wie viele Pixel entsprechen 1 mm in X und Y?
PX_PER_MM_X  = 4.2              # Pixel pro Millimeter in X-Richtung
PX_PER_MM_Y  = 4.2              # Pixel pro Millimeter in Y-Richtung
NOZZLE_PIX_X = IMG_W // 2       # Pixel-Position der Düse im Kamerabild (X)
NOZZLE_PIX_Y = IMG_H // 2       # Pixel-Position der Düse im Kamerabild (Y)

FILL_VOLUME_ML   = 250.0        # Standard-Füllmenge in ml
PUMP_ML_PER_SEC  = 50.0         # Pumpen-Förderrate (ml/s) → einmessen!
# ──────────────────────────────────────────────────────────────────────────────


class BeverageStation:
    """Orchestriert Kamera-Inferenz, Gantry-Bewegung und Befüllung."""

    def __init__(self, model_path: str, port: str, baud: int) -> None:
        print("[INIT] Lade YOLOv8-Modell ...")
        self.model = YOLO(model_path, task="detect")

        print("[INIT] Starte Pi Camera 3 ...")
        self.cam = Picamera2()
        cfg = self.cam.create_video_configuration(
            main={"size": (IMG_W, IMG_H), "format": "RGB888"},
            controls={"FrameRate": 30, "AwbEnable": True},
        )
        self.cam.configure(cfg)
        self.cam.start()
        time.sleep(1.5)

        print(f"[INIT] Verbinde Serial {port} @ {baud} Baud ...")
        self.ser = serial.Serial(port, baud, timeout=2)
        time.sleep(2.5)          # ESP32 startet nach Reset-Puls

        print("[INIT] Homing ...")
        self._gcode("G28")       # Alle Achsen auf Nullpunkt fahren
        print("[OK]  System bereit!\n")

    def _gcode(self, cmd: str) -> str:
        """Sende G-Code an ESP32, warte auf 'ok'-Antwort."""
        self.ser.write((cmd.strip() + "\n").encode("utf-8"))
        response = self.ser.readline().decode("utf-8", errors="replace").strip()
        print(f"  ESP32 ← {cmd:30s}  → {response}")
        return response

    def pixel_to_mm(self, px: float, py: float) -> tuple[float, float]:
        """Wandelt Pixel-Koordinate in Millimeter-Offset zur Düse um."""
        dx = (px - NOZZLE_PIX_X) / PX_PER_MM_X
        dy = (py - NOZZLE_PIX_Y) / PX_PER_MM_Y
        return round(dx, 2), round(dy, 2)

    def detect_bottle(self, frame: np.ndarray) -> dict | None:
        """Führt YOLO-Inferenz durch. Gibt beste Detection zurück oder None."""
        results = self.model(frame, conf=CONF_THRESH, verbose=False)
        boxes   = results[0].boxes
        if len(boxes) == 0:
            return None

        # Beste Detection = höchste Konfidenz
        best_idx  = int(boxes.conf.argmax())
        x1, y1, x2, y2 = boxes.xyxy[best_idx].tolist()
        conf      = float(boxes.conf[best_idx])
        center_x  = (x1 + x2) / 2.0
        center_y  = (y1 + y2) / 2.0

        return {
            "cx":   center_x,
            "cy":   center_y,
            "conf": conf,
            "bbox": (x1, y1, x2, y2),
        }

    def fill(self, detection: dict, volume_ml: float = FILL_VOLUME_ML) -> None:
        """Fährt zur Flaschenöffnung und dosiert das Getränk."""
        dx_mm, dy_mm = self.pixel_to_mm(detection["cx"], detection["cy"])

        print(f"\n[FÜLLEN] Flasche erkannt   conf={detection['conf']:.2f}")
        print(f"         Offset:  Δx={dx_mm:+.1f} mm   Δy={dy_mm:+.1f} mm")

        # Gantry positionieren
        self._gcode("G91")                              # Relative Koordinaten
        self._gcode(f"G1 X{dx_mm} Y{dy_mm} F3000")     # Zur Öffnung fahren
        self._gcode("G90")                              # Zurück zu absolut
        time.sleep(0.2)                                 # Einschwingen lassen

        # Pumpe aktivieren (Custom G-Code M800 = Pumpe an für N ms)
        pump_ms = int((volume_ml / PUMP_ML_PER_SEC) * 1000)
        self._gcode(f"M800 T{pump_ms}")                 # Pumpe EIN
        time.sleep(pump_ms / 1000.0 + 0.3)             # Warten + kleiner Puffer
        self._gcode("M801")                             # Pumpe AUS

        print(f"         {volume_ml:.0f} ml dosiert.")

        # Zurück zur Ausgangsposition (Homing)
        self._gcode("G28")

    def run(self) -> None:
        """Haupt-Loop: wartet auf Flasche und befüllt automatisch."""
        print("[RUN] Warte auf Flasche ...  (Ctrl+C zum Beenden)\n")
        try:
            while True:
                frame     = self.cam.capture_array()
                detection = self.detect_bottle(frame)

                if detection:
                    self.fill(detection)
                    time.sleep(3.0)   # Kurze Pause nach Befüllung
                else:
                    time.sleep(0.02)  # ~50 Hz Polling ohne Detection

        except KeyboardInterrupt:
            print("\n[STOP] Herunterfahren ...")
        finally:
            self.cam.stop()
            self.ser.close()
            print("[OK]  Verbindungen geschlossen.")


if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Smart Beverage Station – Live-Inferenz")
    parser.add_argument("--model",  default=MODEL_PATH,  help="Pfad zum ncnn-Modell-Ordner")
    parser.add_argument("--port",   default=SERIAL_PORT, help="Serial-Port des ESP32")
    parser.add_argument("--volume", type=float, default=FILL_VOLUME_ML, help="Füllmenge in ml")

10X/Y-Gantry – Aufbau

V-Slot-Profile schneiden & verbinden
2× X-Achse, 1× Y-Achse aus 2020er Aluprofil. Eckenverbinder / 3D-gedruckte Halter. Stabile Basis: 2 Querstreben als Tischauflage.
Führungsrollen montieren
Dual V-Slot Rollen auf dem Toolhead-Schlitten. Exzenter-Muttern für spielfreies Spiel einstellen. Ziel: kein Wackeln, aber leichtes Gleiten.
GT2-Riemen spannen
NEMA 17 Motorhalter an den Profilen befestigen. GT2 Zahnriemen über Umlenkrolle führen. Riemenspanner einstellen – der Riemen sollte beim Zupfen kurz schwingen.
TMC2209 verdrahten
UART-Modus: PDN_UART-Pin des TMC2209 via 1 kΩ an TX-Pin des ESP32. VCC: 12 V (Motorspannung), VM: 5 V (Logik). Vref-Strom auf ~0,8 A für NEMA 17 einstellen.
Sensorless Homing testen
StallGuard-Schwelle per SGTHRS-Register einstellen. ESP32-Firmware: bei StallGuard-Interrupt sofort stoppen. Vorsichtig mit niedriger Geschwindigkeit homen.

11Fluidik-System

💧 Kanal 1: Kaltwasser
Mini-Kühlschrank → 12 V Membranpumpe (VEVOR) → Schlauch → Toolhead-Düse. Relais schaltet Pumpe. VL53L0X misst Füllstand im Tank.
🫧 Kanal 2: Sprudel
19 L NC-Keg + CO₂ → Ball-Lock-Anschluss → VEVOR-Pumpe → Düse. CO₂-Druck: ~1 bar. Keg im Kühlschrank lagern für beste Karbonisierung.
🔥 Kanal 3: Heißwasser
CASO HW 660 gehackt: interner Taster via Relais überbrückt (220 V!). Magnetventil DN15 gibt Wasser frei. Nur von Erwachsenen modifizieren!
🍹 Kanal 4: Sirup
12 V Peristaltikpumpe dosiert präzise. Förderrate ~2 ml/s bei 12 V. Kalibrierung: Pumpe 5 s laufen lassen und ml messen → Faktor berechnen.

12Sicherheitskonzept

⚠️
230 V AC im System! Der Heißwasserspender und das Magnetventil arbeiten mit Netzspannung. Diese Teile dürfen nur von Erwachsenen modifiziert und verdrahtet werden. Im Schulprojekt immer unter Aufsicht einer Lehrkraft!
  • 🛑
    Hardware-Notstopp: VL53L0X am Toolhead misst Wasserstand mit 50 Hz. Unterschreitet der Abstand 3 cm (= fast leer / Fehler), löst ein Hardware-Interrupt direkt am ESP32 aus und schaltet alle Relais aus – unabhängig vom Raspberry Pi.
  • 🔌
    Galvanische Trennung: Das 8-Kanal-Relaismodul hat eingebaute Optokoppler-Isolation. 5 V Logik und 12/220 V Lastkreis sind komplett getrennt.
  • 💧
    Wasserspritzschutz: Alle Elektronik-Komponenten sind mindestens 20 cm über Wasserkontakthöhe montiert. Kabelkanäle mit Gefälle nach außen.
  • 🌡️
    Heißwasser-Sicherung: Der CASO HW 660 hat seinen eigenen Thermal-Cutoff. Das Magnetventil (220 V, normal-geschlossen) schließt bei Stromausfall automatisch.
  • 🔧
    Sensorless Homing Limits: Im ESP32-Firmware sind Software-Endstops gesetzt. Die Gantry kann nicht über die physikalischen Grenzen hinausfahren, auch wenn der Nullpunkt falsch ist.
  • 📋
    Jugend forscht Präsentation: Demo läuft immer mit einer Erwachsenen-Aufsicht. Der 230-V-Teil wird im Ausstellungsbetrieb entkoppelt (separater Schalter).

13ESP32 Firmware – Grundgerüst

Der ESP32 empfängt G-Code via USB-Serial und steuert die Stepper-Motoren. Hier ist der relevante C++-Code für TMC2209 UART und StallGuard-Homing.

ℹ️
Bibliotheken: AccelStepper, TMCStepper, Adafruit_VL53L0X. Alle via Arduino Library Manager installierbar.
esp32/firmware/main.cpp 

// Smart Beverage Station — ESP32 Firmware
// Bibliotheken: AccelStepper, TMCStepper, Adafruit_VL53L0X
// IDE: Arduino IDE oder PlatformIO

#include <Arduino.h>
#include <AccelStepper.h>
#include <TMCStepper.h>
#include <Wire.h>
#include "Adafruit_VL53L0X.h"

// ─── Pin-Definitionen ─────────────────────────────────────────────────────
#define X_STEP_PIN    14   // NEMA 17 X-Achse
#define X_DIR_PIN     27
#define X_EN_PIN      26
#define Y_STEP_PIN    25
#define Y_DIR_PIN     33
#define Y_EN_PIN      32

#define TMC_SERIAL    Serial2   // UART für TMC2209
#define TMC_RX        16
#define TMC_TX        17
#define R_SENSE       0.11f    // Shunt-Widerstand des TMC2209

#define RELAY_PIN_0   4        // Relais 1–4 (Kaltwasser, Sprudel, Heißwasser, Sirup)
#define RELAY_PIN_1   5
#define RELAY_PIN_2   18
#define RELAY_PIN_3   19

#define WATER_SENSOR_INTERRUPT 23  // Hardware-Interrupt von VL53L0X
#define WATER_THRESHOLD_MM     30  // Notstopp unter 30 mm Abstand

// ─── Globale Objekte ───────────────────────────────────────────────────────
AccelStepper stepperX(AccelStepper::DRIVER, X_STEP_PIN, X_DIR_PIN);
AccelStepper stepperY(AccelStepper::DRIVER, Y_STEP_PIN, Y_DIR_PIN);
TMC2209Stepper driverX(&TMC_SERIAL, R_SENSE, 0b00);  // ADDR=0
TMC2209Stepper driverY(&TMC_SERIAL, R_SENSE, 0b01);  // ADDR=1
Adafruit_VL53L0X lox;

volatile bool emergencyStop = false;
String gCodeBuffer = "";

// ─── Notstopp ISR ─────────────────────────────────────────────────────────
void IRAM_ATTR onWaterLow() {
  emergencyStop = true;  // Im ISR nur Flag setzen!
}

// ─── TMC2209 Initialisierung ───────────────────────────────────────────────
void initDrivers() {
  TMC_SERIAL.begin(115200, SERIAL_8N1, TMC_RX, TMC_TX);

  for (auto* d : {&driverX, &driverY}) {
    d->begin();
    d->toff(4);           // Treiber aktivieren
    d->blank_time(24);
    d->rms_current(800);  // 800 mA RMS für NEMA 17
    d->microsteps(16);    // 16 Mikroschritt
    d->en_spreadCycle(false); // StealthChop aktivieren (leiser)
    d->SGTHRS(100);       // StallGuard-Schwelle (0-255, höher = empfindlicher)
  }
}

// ─── Sensorless Homing ────────────────────────────────────────────────────
void homeAxis(AccelStepper& stepper, int enPin, TMC2209Stepper& driver) {
  digitalWrite(enPin, LOW);
  stepper.setMaxSpeed(200);   // Langsam homen!
  stepper.setAcceleration(100);
  stepper.move(-10000);       // Fahre Richtung Endstop

  while (stepper.isRunning()) {
    stepper.run();
    if (driver.SG_RESULT() < 50) {  // StallGuard ausgelöst = Endstop erreicht
      stepper.stop();
      stepper.setCurrentPosition(0);
      break;
    }
  }
  stepper.setMaxSpeed(1000);
  stepper.setAcceleration(500);
}

// ─── G-Code Parser ────────────────────────────────────────────────────────
void parseGCode(String line) {
  line.trim();
  if (line.length() == 0 || line[0] == ';') return;

  if (line.startsWith("G28")) {          // Homing
    homeAxis(stepperX, X_EN_PIN, driverX);
    homeAxis(stepperY, Y_EN_PIN, driverY);

  } else if (line.startsWith("G1")) {    // Lineare Bewegung
    float x = 0, y = 0, f = 1000;
    sscanf(line.c_str(), "G1 X%f Y%f F%f", &x, &y, &f);
    stepperX.moveTo((long)(x * 80));     // 80 Steps/mm bei 16 Mikro + GT2
    stepperY.moveTo((long)(y * 80));
    stepperX.setMaxSpeed(f);
    stepperY.setMaxSpeed(f);

  } else if (line.startsWith("M800")) {  // Custom: Pumpe AN für T ms
    int t = 0;
    sscanf(line.c_str(), "M800 T%d", &t);
    digitalWrite(RELAY_PIN_0, HIGH);
    delay(t);
    digitalWrite(RELAY_PIN_0, LOW);

  } else if (line.startsWith("M801")) {  // Custom: Pumpe AUS
    digitalWrite(RELAY_PIN_0, LOW);
  }

  Serial.println("ok");  // Bestätigung an Raspberry Pi
}

// ─── Setup ────────────────────────────────────────────────────────────────
void setup() {
  Serial.begin(115200);
  Wire.begin();

  // Relais-Pins
  for (int pin : {RELAY_PIN_0, RELAY_PIN_1, RELAY_PIN_2, RELAY_PIN_3}) {
    pinMode(pin, OUTPUT);
    digitalWrite(pin, LOW);
  }

  // Stepper Enable-Pins
  pinMode(X_EN_PIN, OUTPUT); digitalWrite(X_EN_PIN, HIGH);  // Erst deaktiviert
  pinMode(Y_EN_PIN, OUTPUT); digitalWrite(Y_EN_PIN, HIGH);

  // TMC2209 Treiber
  initDrivers();

  // VL53L0X Laser-Sensor
  if (!lox.begin()) {
    Serial.println("[ERROR] VL53L0X nicht gefunden!");
  }

  // Hardware-Interrupt für Notstopp
  pinMode(WATER_SENSOR_INTERRUPT, INPUT_PULLUP);
  attachInterrupt(digitalPinToInterrupt(WATER_SENSOR_INTERRUPT),
                  onWaterLow, FALLING);

  Serial.println("[OK] Smart Beverage Station bereit.");
}

// ─── Loop ─────────────────────────────────────────────────────────────────
void loop() {
  // Notstopp überprüfen
  if (emergencyStop) {
    stepperX.stop();
    stepperY.stop();
    for (int pin : {RELAY_PIN_0, RELAY_PIN_1, RELAY_PIN_2, RELAY_PIN_3})
      digitalWrite(pin, LOW);
    Serial.println("[EMERGENCY] Notstopp ausgelöst! Wasserstand kritisch.");
    while (true) delay(1000);  // Halte an bis Reset
  }

  // G-Code einlesen
  while (Serial.available()) {
    char c = Serial.read();
    if (c == '\n') {
      parseGCode(gCodeBuffer);
      gCodeBuffer = "";
    } else {
      gCodeBuffer += c;
    }
  }

  // Stepper laufen lassen
  stepperX.run();
  stepperY.run();
}
